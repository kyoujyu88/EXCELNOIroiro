Sub AddToSharePointWithSQL()
    Dim conn As Object
    Dim dbPath As String
    Dim tableName As String
    Dim strSQL As String
    
    ' 1. 同じフォルダにあるAccessファイルを指定します
    dbPath = ThisWorkbook.Path & "\SP_Link.accdb" 
    
    ' 2. Accessの中に表示されているリンクテーブルの名前を指定します
    tableName = "TestList" 
    
    Set conn = CreateObject("ADODB.Connection")
    
    ' Accessのファイルに接続します
    conn.Open "Provider=Microsoft.ACE.OLEDB.12.0;Data Source=" & dbPath & ";"
    
    ' SQL文を作りますっ
    ' （※列の名前は、Accessのテーブル上の名前に合わせてくださいね）
    strSQL = "INSERT INTO [" & tableName & "] ([Title], [本文]) " & _
             "VALUES ('追加するタイトルです', 'ここに本文の内容を入れますっ')"
    
    ' SQL文を実行して、直接データを追加します！
    conn.Execute strSQL
    
    ' 閉じてお片付けします
    conn.Close
    Set conn = Nothing
    
    MsgBox "データの追加が完了しました…！"
End Sub
