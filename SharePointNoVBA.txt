Sub AddToSharePointWithDAO()
    Dim dbEngine As Object
    Dim db As Object
    Dim rs As Object
    Dim dbPath As String
    Dim tableName As String
    
    ' 1. Accessファイルの場所を指定します
    dbPath = ThisWorkbook.Path & "\SP_Link.accdb" 
    
    ' 2. Accessの中に表示されているリンクテーブルの名前を指定します
    tableName = "TestList" 
    
    ' DAOという、Access専用の強力な連絡係を呼び出しますっ
    Set dbEngine = CreateObject("DAO.DBEngine.120")
    
    ' Accessのファイルを開きます
    Set db = dbEngine.OpenDatabase(dbPath)
    
    ' テーブルを開きます（dbOpenDynasetという開き方になります）
    Set rs = db.OpenRecordset(tableName, 2) ' 2 = dbOpenDynaset
    
    ' 新しいアイテムを追加します…！
    rs.AddNew
    
    ' 列の名前と、追加したい内容を指定します
    rs.Fields("Title").Value = "DAOで追加するタイトルです"
    
    ' 本文の列名を指定します（※Access上の列名に合わせてくださいね）
    rs.Fields("本文").Value = "今度こそ、本文も無事に届きますように…！"
    
    ' 変更を保存します
    rs.Update
    
    ' 閉じてお片付けします
    rs.Close
    db.Close
    Set rs = Nothing
    Set db = Nothing
    Set dbEngine = Nothing
    
    MsgBox "データの追加が完了しましたっ…！"
End Sub
