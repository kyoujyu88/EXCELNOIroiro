Sub AddSpecificColumnsToSharePoint()
    Dim conn As Object
    Dim rs As Object
    Dim siteUrl As String
    Dim listName As String
    
    ' 接続先の情報を設定します
    siteUrl = "https://yourtenant.sharepoint.com/sites/yoursite"
    listName = "{リストのGUID}"
    
    Set conn = CreateObject("ADODB.Connection")
    Set rs = CreateObject("ADODB.Recordset")
    
    ' SharePointリストへ接続します
    conn.Open "Provider=Microsoft.ACE.OLEDB.12.0;WSS;IMEX=0;RetrieveIds=Yes;DATABASE=" & siteUrl & ";LIST=" & listName & ";"
    
    ' レコードセットを開きます
    rs.Open "SELECT * FROM [" & listName & "]", conn, 1, 3
    
    ' 新しいアイテムを追加します…！
    rs.AddNew
    
    ' 1. タイトル
    rs.Fields("Title").Value = "追加するタイトルです"
    
    ' 2. 本文 (※ "Body" の部分は実際の内部名に書き換えてください)
    rs.Fields("Body").Value = "ここに追加したい本文の内容を入力します。"
    
    ' 3. 有効期限 (※ "ExpirationDate" の部分は実際の内部名に書き換えてください)
    ' 日付は #年/月/日# のように囲むか、Excelのセルの値を指定します
    rs.Fields("ExpirationDate").Value = #12/31/2026# 
    
    ' 変更を保存します
    rs.Update
    
    ' 閉じてお片付けします
    rs.Close
    conn.Close
    Set rs = Nothing
    Set conn = Nothing
    
    MsgBox "リストへの追加が完了しました…！"
End Sub
